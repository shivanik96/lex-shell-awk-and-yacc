%{ 
#include<stdio.h> 
%} 
%token NAME NUMBER 
%% 
Statement : NAME ‘=’ expr  ; 
   | expr 
expr  : expr ‘+’ NUMBER { $$=$1+$3; 
      printf(“Ans = %d”, $$); 
   | expr ‘-‘ NUMBER { $$=$1-$3; 
      printf(“Ans = %d”, $$);  
   | expr ‘*‘ NUMBER { $$=$1*$3; 
      printf(“Ans = %d”, $$); 
| expr ‘/‘ NUMBER { $$=$1/$3; 
      printf(“Ans = %d”, $$);  
| NUMBER 
   ; 
%% 
Int main() 
{ 
Return yyparse(); 
} 
Int yyerror(char* msg) 
{ 
Return printf(“%s”,msg); 
}
